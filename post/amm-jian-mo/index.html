AMM 建模

用户行为定义

AMM可以建模为一组token的组合：

Pool={T0,T1,...,Ti}Reservers={r0,r1,...,ri}totalSupplyPool = \{T_0, T_1, ... ,T_i\} \\ Reservers = \{r_0, r_1, ... ,r_i\} \\ totalSupply Pool={T0​,T1​,...,Ti​}Reservers={r0​,r1​,...,ri​}totalSupply

由此定义用户操作包括Transfer，AddLiquidity，RemoveLiquidity，Swap：

Swap(Δr0,Δr1)=>r0+Δr0,r1+Δr1Remove/AddLiquidity(Δr0,Δr1,ΔtotalSupply)=>r0+Δr0,r1+Δr1,totalSupply+ΔtotalSupplyTransfer(Δri)=>ri+Δri(Δri>0)Swap(\Delta r_0, \Delta r_1 ) => r_0+\Delta r_0, r_1+\Delta r_1 \\ Remove/AddLiquidity(\Delta r_0,\Delta r_1, \Delta totalSupply ) => r_0+\Delta r_0, r_1+\Delta r_1, totalSupply + \Delta totalSupply \\ Transfer(\Delta r_i ) => r_i + \Delta r_i (\Delta r_i >0 ) Swap(Δr0​,Δr1​)=>r0​+Δr0​,r1​+Δr1​Remove/AddLiquidity(Δr0​,Δr1​,ΔtotalSupply)=>r0​+Δr0​,r1​+Δr1​,totalSupply+ΔtotalSupplyTransfer(Δri​)=>ri​+Δri​(Δri​>0)

Swap

Swap函数主要分为两类，一种为单边兑换，即输入一种代币，同时兑换出另一种代币，以Uniswap v2为例：

     function swap(uint amount0Out, uint amount1Out, address to, bytes calldata data) external lock {
        (uint112 _reserve0, uint112 _reserve1,) = getReserves(); 
        ...

        address _token1 = token1;
        require(to != _token0 && to != _token1, 'UniswapV2: INVALID_TO');

        if (amount1Out > 0) _safeTransfer(_token1, to, amount1Out);

        balance1 = IERC20(_token1).balanceOf(address(this));
        }
        uint amount0In = balance0 > _reserve0 - amount0Out ? balance0 - (_reserve0 - amount0Out) : 0;
        uint amount1In = balance1 > _reserve1 - amount1Out ? balance1 - (_reserve1 - amount1Out) : 0;
        require(amount0In > 0 || amount1In > 0, 'UniswapV2: INSUFFICIENT_INPUT_AMOUNT');
        { 
        uint balance0Adjusted = balance0.mul(1000).sub(amount0In.mul(3));
        uint balance1Adjusted = balance1.mul(1000).sub(amount1In.mul(3));

        }
        _update(balance0, balance1, _reserve0, _reserve1);
        ...
    }

   function joinPool(uint poolAmountOut, uint[] calldata maxAmountsIn)
    {
        uint poolTotal = totalSupply();
        uint ratio = bdiv(poolAmountOut, poolTotal);
        for (uint i = 0; i < _tokens.length; i++) {
            address t = _tokens[i];
            uint bal = _records[t].balance;
            uint tokenAmountIn = bmul(ratio, bal);
            require(tokenAmountIn != 0, "ERR_MATH_APPROX");
            require(tokenAmountIn <= maxAmountsIn[i], "ERR_LIMIT_IN");
            _records[t].balance = badd(_records[t].balance, tokenAmountIn);
            _pullUnderlying(t, msg.sender, tokenAmountIn);
        }
        _mintPoolShare(poolAmountOut);
        _pushPoolShare(msg.sender, poolAmountOut);
    }

    function exitPool(uint poolAmountIn, uint[] calldata minAmountsOut)
    {
        uint poolTotal = totalSupply();
        uint exitFee = bmul(poolAmountIn, EXIT_FEE);
        uint pAiAfterExitFee = bsub(poolAmountIn, exitFee);
        uint ratio = bdiv(pAiAfterExitFee, poolTotal);
        require(ratio != 0, "ERR_MATH_APPROX");
        _pullPoolShare(msg.sender, poolAmountIn);
        _pushPoolShare(_factory, exitFee);
        _burnPoolShare(pAiAfterExitFee);
        for (uint i = 0; i < _tokens.length; i++) {
            address t = _tokens[i];
            uint bal = _records[t].balance;
            uint tokenAmountOut = bmul(ratio, bal);
            require(tokenAmountOut != 0, "ERR_MATH_APPROX");
            require(tokenAmountOut >= minAmountsOut[i], "ERR_LIMIT_OUT");
            _records[t].balance = bsub(_records[t].balance, tokenAmountOut);
            emit LOG_EXIT(msg.sender, t, tokenAmountOut);
            _pushUnderlying(t, msg.sender, tokenAmountOut);
        }
    }
